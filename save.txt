const proyectsContainer = document.querySelector('.swiper');

function cardProyect(proyects, limit = proyects.length) {
    const cardsWrapper = document.createElement('div');
    cardsWrapper.classList = 'swiper-wrapper'
    cardsWrapper.style.display = 'flex';
    cardsWrapper.style.flexWrap = 'wrap';
    cardsWrapper.style.justifyContent = 'center';
    cardsWrapper.style.gap = '20px'; // Espacio entre tarjetas
    
    proyects.slice(0, limit).forEach(proyect => {
        const newProyect = document.createElement('div');
        newProyect.classList = 'swiper-slide';
        newProyect.innerHTML = `
            <div class="card-projet"> <!-- Este div debería tener clase swiper, es correcto -->
                        <img src='${proyect.img}'>
                        <div class="card__content">
                            <p class="card__title">${proyect.name}</p>
                            <p class="card__description">${proyect.description}</p>
                            <button class="card__button">Live Demo</button>
                            <button class="card__button secondary">Code</button>
                            <div class="categories">
                                <span>HTML</span>
                                <span>CSS</span>
                                <span>JAVASCRIPT</span>
                            </div>
                        </div> <!-- Cierre de card__content -->
            </div> <!-- Cierre de card-projet -->
        `;
        cardsWrapper.appendChild(newProyect);
    });
    proyectsContainer.appendChild(cardsWrapper);
}
